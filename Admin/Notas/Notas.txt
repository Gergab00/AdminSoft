/*Creación de base de datos*/
CREATE DATABASE adminsoft CHARACTER SET utf8 COLLATE utf8_general_ci;
/*Creacion de tabla Departamentos*/
CREATE TABLE `adminsoft`.`departamentos` ( `ID` VARCHAR(100) NOT NULL , `NOMDEPTO` VARCHAR(200) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
/*Creacion de tabla Grupos*/
CREATE TABLE `adminsoft`.`grupos` ( `ID` VARCHAR(100) NOT NULL , `NOMGPO` VARCHAR(200) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
/*Creacion de tabla Insumos*/
CREATE TABLE `adminsoft`.`insumos` ( `ID` VARCHAR(100) NOT NULL, `NOMINSUMO` VARCHAR(200) NOT NULL, `NOMDEPTO` VARCHAR(200) NOT NULL , `UNIDAD` VARCHAR(100) NOT NULL , `PRECIO_S_IVA` DOUBLE(10,2) NOT NULL , `TIENE_IVA` SET('true','false') NOT NULL , `IVA_PORC` DOUBLE(10,2) NOT NULL , `IVA_VAL` DOUBLE(10,2) NOT NULL , `PRECIO_C_IVA` DOUBLE(10,2) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
/*Creación de tabla Presentaciones*/
CREATE TABLE `adminsoft`.`presentaciones` ( `ID` VARCHAR(100) NOT NULL , `NOMPRES` VARCHAR(200) NOT NULL , `NOMINSUMO` VARCHAR(200) NOT NULL , `VOLUMEN` VARCHAR(200) NOT NULL , `UNIDAD` VARCHAR(200) NOT NULL , `PRECIOS/IVA` DOUBLE(10,2) NOT NULL , `TIENEIVA` SET('true','false') NOT NULL , `IVA_VAL` DOUBLE(10,2) NOT NULL , `IVA_PORC` DOUBLE(10,2) NOT NULL , `PRECIOC/IVA` DOUBLE(10,2) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
/*Creación de tabla Productos*/
CREATE TABLE `adminsoft`.`productos` ( `IDPRODUCTO` VARCHAR(100) NOT NULL , `NOMPRODUCTO` VARCHAR(200) NOT NULL , `NOMGRUPO` VARCHAR(200) NOT NULL , `PRECIOSIVA` DOUBLE(10,2) NOT NULL , `TIENEIVA` SET('true','false') NOT NULL , `IVAPORC` DOUBLE(10,2) NOT NULL , `IVAVAL` DOUBLE(10,2) NOT NULL , `PRECIOCIVA` DOUBLE(10,2) NOT NULL , PRIMARY KEY (`IDPRODUCTO`)) ENGINE = MyISAM;
